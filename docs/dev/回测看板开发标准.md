### 4.3 API 调用

```javascript
async function run() {
  const btn = document.getElementById('btn-run');
  btn.disabled = true; btn.textContent = '⏳';

  // 查询名称
  const code = document.getElementById('stock-code').value;
  try {
    const nr = await fetch(API_BASE + '/api/stock-name?code=' + code);
    if (nr.ok) {
      const nd = await nr.json();
      document.getElementById('stock-name').textContent = nd.name || '';
    }
  } catch (e) {}

  // 主API
  try {
    const r = await fetch(API_BASE + '/api/signal-type', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        stock_code: code,
        start: document.getElementById('date-start').value,
        end: document.getElementById('date-end').value,
        period: document.getElementById('period').value,
        params: gp()
      })
    });
    const d = await r.json();
    ck = d.klines || []; cs = d.signals || [];
    renderChart(); renderTable();
    btn.textContent = '🔄';
  } catch (e) { btn.textContent = '❌'; }
  btn.disabled = false;
}
```

### 4.5 配置持久化

```javascript
const SIGNAL_TYPE = 'signal_type_name';

async function loadCfg() {
  const r = await fetch(API_BASE + '/api/config?signal_type=' + SIGNAL_TYPE, { cache: 'no-store' });
  const c = await r.json();
  if (!c || !c.signal_type) return;
  const b = c.signal_type;
  svs('p-xxx', Math.round(b.param * 100));
  stog('p-xxx-on', b.param_enabled !== false);
}

async function saveCfg() {
  const p = gp().signal_type;
  const y = ['# 标题', 'signal_type:',
    '  param: ' + p.param, '  param_enabled: ' + (p.param_enabled ? 'true' : 'false'), ''
  ].join('\n');
  await fetch(API_BASE + '/api/config?signal_type=' + SIGNAL_TYPE, {
    method: 'POST', headers: { 'Content-Type': 'text/plain' }, body: y
  });
}

function svs(id, v) { const e = document.getElementById(id); if (!e) return; e.value = v; e.dispatchEvent(new Event('input', { bubbles: true })); }
function stog(id, on) { const b = document.getElementById(id); if (!b) return; b.className = 'tsw ' + (on ? 'on' : 'off'); }
```

## 五、排查功能规范

每个回测看板必须包含排查工具，位于信号明细卡片下方。

### 5.1 HTML 模板

```html
<div class="xhs-card"><div class="xhs-card-header"><span class="xhs-card-label">🔎 单日排查</span></div>
  <div style="padding:12px 16px;display:flex;gap:8px;align-items:center">
    <input type="date" id="diag-date" style="...">
    <button onclick="runDiag()">排查</button>
  </div>
  <div id="diag-result" style="padding:0 16px 14px;display:none"></div>
</div>
```

### 5.2 设计原则

1. **排查项必须覆盖引擎所有过滤条件**，不能漏任何一项
2. 每一项显示：条件名、实际值、阈值、✅/❌判定
3. 引擎的 toggle 开关状态要在排查中体现（关掉的检查标记为"(已关闭)"）
4. 排查数据从当前加载的 `ck`（K线数组）计算，无需请求服务器

### 5.3 示例

```javascript
function runDiag() {
  const date = document.getElementById('diag-date').value;
  const div = document.getElementById('diag-result');
  // ... 从 ck 中找日期索引, 计算各项条件, 输出表格
  function cc(v,t){return v?'<td style=color:#4CAF50>✅ '+t+'</td>':'<td style=color:#FF6B6B>❌ '+t+'</td>';}
  div.innerHTML = '<table>...<tr><td>条件</td><td>实际值</td><td>阈值</td><td>判定</td></tr>...</table>';
  div.style.display = 'block';
}
```

  "stock_code": "600519",
